#include <WiFi.h>
#include <PubSubClient.h>
#include "DHT.h"

// ====== ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ Wi-Fi ======
const char* ssid ="MEESUKFloor2";     // ‡πÉ‡∏™‡πà SSID WiFi ‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì
const char* password = "999999999"; // ‡πÉ‡∏™‡πà‡∏£‡∏´‡∏±‡∏™‡∏ú‡πà‡∏≤‡∏ô WiFi ‡∏Ç‡∏≠‡∏á‡∏Ñ‡∏∏‡∏ì

// ====== ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ MQTT ======
const char* mqtt_server = "broker.hivemq.com";
const int mqtt_port = 1883;   // ‚úÖ ‡πÉ‡∏ä‡πâ TCP port ‡πÑ‡∏°‡πà‡πÉ‡∏ä‡πà 8000
const char* mqtt_client_id = "ESP32Client"; // ‡∏ä‡∏∑‡πà‡∏≠ client (‡∏ï‡πâ‡∏≠‡∏á‡πÑ‡∏°‡πà‡∏ã‡πâ‡∏≥‡∏Å‡∏±‡∏ô)

// ====== ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ DHT11 ======
#define DHTPIN 4
#define DHTTYPE DHT11
DHT dht(DHTPIN, DHTTYPE);

// ====== ‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤ LDR ======
#define LDR_PIN 34

WiFiClient espClient;
PubSubClient client(espClient);

void setup_wifi() {
  delay(10);
  Serial.println("Connecting to WiFi...");
  WiFi.begin(ssid, password);
  while (WiFi.status() != WL_CONNECTED) {
    delay(500);
    Serial.print(".");
  }
  Serial.println("\n‚úÖ WiFi connected!");
  Serial.print("IP address: ");
  Serial.println(WiFi.localIP());
}

void reconnect() {
  while (!client.connected()) {
    Serial.print("Connecting to MQTT...");
    if (client.connect(mqtt_client_id)) {
      Serial.println("‚úÖ connected to MQTT broker");
    } else {
      Serial.print("‚ùå failed, rc=");
      Serial.print(client.state());
      Serial.println(" -> try again in 5 seconds");
      delay(5000);
    }
  }
}

void setup() {
  Serial.begin(115200);
  dht.begin();
  setup_wifi();
  client.setServer(mqtt_server, mqtt_port);
}

void loop() {
  if (!client.connected()) {
    reconnect();
  }
  client.loop();

  // ‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡πà‡∏≤‡πÄ‡∏ã‡∏ô‡πÄ‡∏ã‡∏≠‡∏£‡πå
  float temperature = dht.readTemperature();
  float humidity = dht.readHumidity();
  int light = analogRead(LDR_PIN);

  // ‡πÄ‡∏ä‡πá‡∏Ñ‡∏ß‡πà‡∏≤‡∏≠‡πà‡∏≤‡∏ô‡∏Ñ‡πà‡∏≤ DHT ‡∏™‡∏≥‡πÄ‡∏£‡πá‡∏à
  if (isnan(temperature) || isnan(humidity)) {
    Serial.println("‚ùå Failed to read from DHT sensor!");
    return;
  }

  // ‡∏™‡∏£‡πâ‡∏≤‡∏á JSON payload
  char payload[120];
  snprintf(payload, sizeof(payload),
           "{\"temperature\": %.2f, \"humidity\": %.2f, \"light\": %d}",
           temperature, humidity, light);

  // ‡∏™‡πà‡∏á‡∏Ñ‡πà‡∏≤‡πÑ‡∏õ MQTT
  client.publish("home/sensor/data", payload);

  Serial.print("üì§ Sent: ");    
  Serial.println(payload);

  delay(5000); // ‡∏™‡πà‡∏á‡∏ó‡∏∏‡∏Å 5 ‡∏ß‡∏¥‡∏ô‡∏≤‡∏ó‡∏µ
}
